pipeline {
    agent any

    stages {
      stage('Verify Branch') {
         steps {
            echo $GIT_BRANCH
         }
      }
    }
    stages {
        stage ("Build") {
            steps {
                sh "mvn -version"
                sh "mvn clean install"
            }
        }
    }
    post {
        always {
            cleanWs()
        }
    }
}
